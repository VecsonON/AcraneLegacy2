    immediate = {

            create_random_arcane_artifact = yes
            scope:al_artifact_found = { save_scope_as = al_artifact_1_found }
            create_random_arcane_artifact = yes
            scope:al_artifact_found = { save_scope_as = al_artifact_2_found }
            create_random_arcane_artifact = yes
            scope:al_artifact_found = { save_scope_as = al_artifact_3_found }
            create_random_arcane_artifact = yes
            scope:al_artifact_found = { save_scope_as = al_artifact_4_found }
            set_variable = created_artifacts_al
        }
    }
	option = { # Take 1
		name = al_activities_events.0042.a
        trigger = { exists = scope:al_ancient_tome_found NOT = { has_variable = al_ancient_tome_found_get } }
        scope:al_ancient_tome_found = { set_owner = root }
        set_variable = al_ancient_tome_found_get
        hidden_effect = {
            random_list = {
                25 = { trigger_event = al_activities_events.0142 }
                75 = { trigger_event = al_activities_events.0042 }
            }
        }
	}
	option = { # Take 2
		name = al_activities_events.0042.b
        trigger = { exists = scope:al_artifact_1_found NOT = { has_variable = al_artifact_1_found_get } }
        scope:al_artifact_1_found = { set_owner = root }
        set_variable = al_artifact_1_found_get
        hidden_effect = {
            random_list = {
                25 = { trigger_event = al_activities_events.0142 }
                75 = { trigger_event = al_activities_events.0042 }
            }
        }
	}
	option = { # Take 3
		name = al_activities_events.0042.c
        trigger = { exists = scope:al_artifact_2_found NOT = { has_variable = al_artifact_2_found_get } }
        scope:al_artifact_2_found = { set_owner = root }
        set_variable = al_artifact_2_found_get
        hidden_effect = {
            random_list = {
                25 = { trigger_event = al_activities_events.0142 }
                75 = { trigger_event = al_activities_events.0042 }
            }
        }
	}
	option = { # Take 4
		name = al_activities_events.0042.d
        trigger = { exists = scope:al_artifact_3_found NOT = { has_variable = al_artifact_3_found_get } }
        scope:al_artifact_3_found = { set_owner = root }
        set_variable = al_artifact_3_found_get
        hidden_effect = {
            random_list = {
                25 = { trigger_event = al_activities_events.0142 }
                75 = { trigger_event = al_activities_events.0042 }
            }
        }
	}
	option = { # Take 5
		name = al_activities_events.0042.e
        trigger = { exists = scope:al_artifact_4_found NOT = { has_variable = al_artifact_4_found_get } }
        scope:al_artifact_4_found = { set_owner = root }
        set_variable = al_artifact_4_found_get
        hidden_effect = {
            random_list = {
                25 = { trigger_event = al_activities_events.0142 }
                75 = { trigger_event = al_activities_events.0042 }
            }
        }
	}
    option = { # Ignore
		name = al_activities_events.0042.f
        remove_variable = created_artifacts_al
        hidden_effect = {
            if = { limit = { has_variable = al_ancient_tome_found_get } remove_variable = al_ancient_tome_found_get }
            else = { character:AL_ancient = { destroy_artifact = scope:al_ancient_tome_found } }
            if = { limit = { has_variable = al_artifact_1_found_get } remove_variable = al_artifact_1_found_get }

	}
